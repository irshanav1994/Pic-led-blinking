#include<xc.h>
char receive(void);
void trans(unsigned char a );
void main(void) {
    TRISD = 0X00;
    TRISC = 0X80;
    TXSTA = 0X24;
    RCSTA = 0X90;
    SPBRG = 129; 
    
char w;

    int flag = 0;  // Flag to control transmission
    while(1)
    {
        w = receive();
        if (w == '*') {
            flag = 1;  // Start transmission when '*' is received
        } 
        else if (w == '#')
        {
           flag = 0;  // Stop transmission when '#' is received
        } 
        else if (flag)
        {
            trans(w);  // Only transmit if within '*' and '#'
        }
    }
}
